/**
 * This is the Board class that stores the state of the othello game board.
 */
class Board {
    /**
     * Creates an instance of Board.
     * @param {number} size The size of the board (must be even and greater than or equal to 4).
     * @param {string[]} [state] The initial state to set the board.
     * @throws {InvalidBoardSize} Thrown if the specified board size is not even or less than 4.
     * @throws {InvalidStateSize} Thrown if the state size is not the square of the board size.
     * @throws {InvalidCellValue} Thrown is the state contains a cell where the value is neither of {'B', 'W', 'N'}
     */
    constructor(size, state) {
        if (size % 2 !== 0 || size < 4)
            throw "InvalidBoardSize";
        if (state !== undefined && size * size !== state.length)
            throw "InvalidStateSize";
        if (state !== undefined)
            for (let i = 0; i < state.length; i++)
                if (!this.validValue(state[i]))
                    throw "InvalidCellValue";
 
        /**
         * This variable stores the size of the board (the board is a square grid of size-by-size).
         */
        this.size = size;
         /**
         * This variable stores the state of the individual cells of the board.
         * @property {string[]} state [state]{@link Board#state}
         * @property {number} size [size]{@link Board#size}
         */
        this.state = [];
        
        if (state !== undefined)
            this.state = JSON.parse(JSON.stringify(state));
 
        if (state === undefined) {
            this.state = [...Array(64)].fill('N');
 
            let tl = size / 2 - 1;
            let br = size / 2;
 
            this.state[this.index(tl, tl)] = 'W';
            this.state[this.index(br, br)] = 'W';
            this.state[this.index(tl, br)] = 'B';
            this.state[this.index(br, tl)] = 'B';
        }
 
        this.listeners = [];
    }
 
    /**
     * Returns the size of the board.
     */
    getBoardSize() {
        return this.size;
    }
 
    /**
     * Returns true if the (x, y) cell is within the bounds of the board, otherwise false.
     * @param {number} x The x location of the cell
     * @param {number} y The y location of the cell
     * @returns {boolean} <code>true</code> if the (x, y) cell is within the bounds of the board, otherwise <code>false</code>.
     */
    isCellInBoard(x, y) {
        return x >= 0 && y >= 0 && x < this.size && y < this.size;
    }
 
    /**
     * Checks if a cell is set to a particular value.
     * @param {number} x The x location of the cell
     * @param {number} y The y location of the cell
     * @param {string} v The value to test against
     * @throws {InvalidCellLocation} Thrown if the (x, y) cell doesn't exist
     * @throws {InvalidCellValue} Thrown if the specified value is neither of {'B', 'W', 'N'}
     * @returns {boolean} <code>true</code> if the (x, y) cell is set to the specified value, otherwise <code>false</code>.
     */
    isCellValue(x, y, v) {
        if (!this.isCellInBoard(x, y))
            throw "InvalidCellLocation";
        if (!this.validValue(v))
            throw "InvalidCellValue";
 
        return this.state[this.index(x, y)] === v;
    }
 
    /**
     * Checks if the cell is a valid move for a specified player.
     * @param {number} x The x location of the cell
     * @param {number} y The y location of the cell
     * @param {string} player The player to test for a valid move
     * @throws {InvalidCellLocation} Thrown if the (x, y) cell doesn't exist
     * @throws {InvalidPlayer} Thrown if the specfied player is neither of {'B', 'W'}
     * @returns {number} A positive value representing possible increase in score and a valid move, otherwise 0 representing invalid move.
     */
    valid(x, y, player) {
        if (!this.isCellInBoard(x, y))
            throw "InvalidCellLocation";
        if (!this.validPlayer(player))
            throw "InvalidPlayer";
 
        // The index of the cell.
        let idx = this.index(x, y);
 
        // If the cell is not empty, it is an invalid move.
        if (this.state[idx] !== 'N')
            return 0;
 
        // The opponent of the player.
        let opponent = player === 'B' ? 'W' : 'B';
 
        // The number of opponent cells in each direction.
        let left = 0;
        let right = 0;
        let top = 0;
        let bottom = 0;
        let topleft = 0;
        let topright = 0;
        let bottomleft = 0;
        let bottomright = 0;
 
        // The x and y coordinates to keep track of the cell that is being inspected for each direction.
        let xil = x - 1;
        let xir = x + 1;
        let yit = y - 1;
        let yib = y + 1;
        let xitl = x - 1;
        let yitl = y - 1;
        let xitr = x + 1;
        let yitr = y - 1;
        let xibl = x - 1;
        let yibl = y + 1;
        let xibr = x + 1;
        let yibr = y + 1;
 
        // Count the amount of opponent cells in each direction.
        for (; this.isCellInBoard(xil, y) && this.isCellValue(xil, y, opponent); xil--) left++;
        for (; this.isCellInBoard(x, yit) && this.isCellValue(x, yit, opponent); yit--) top++;
        for (; this.isCellInBoard(xir, y) && this.isCellValue(xir, y, opponent); xir++) right++;
        for (; this.isCellInBoard(x, yib) && this.isCellValue(x, yib, opponent); yib++) bottom++;
        for (; this.isCellInBoard(xitr, yitr) && this.isCellValue(xitr, yitr, opponent); xitr++ , yitr--) topright++;
        for (; this.isCellInBoard(xitl, yitl) && this.isCellValue(xitl, yitl, opponent); xitl-- , yitl--) topleft++;
        for (; this.isCellInBoard(xibr, yibr) && this.isCellValue(xibr, yibr, opponent); xibr++ , yibr++) bottomright++;
        for (; this.isCellInBoard(xibl, yibl) && this.isCellValue(xibl, yibl, opponent); xibl-- , yibl++) bottomleft++;
 
        let total = 0;
 
        // Check if there is at least 1 opponent cell in the respective directions and ends with a player cell.
        if (this.isCellInBoard(xil, y) && this.isCellValue(xil, y, player) && left > 0) total += left;
        if (this.isCellInBoard(xir, y) && this.isCellValue(xir, y, player) && right > 0) total += right;
        if (this.isCellInBoard(x, yit) && this.isCellValue(x, yit, player) && top > 0) total += top;
        if (this.isCellInBoard(x, yib) && this.isCellValue(x, yib, player) && bottom > 0) total += bottom;
        if (this.isCellInBoard(xitr, yitr) && this.isCellValue(xitr, yitr, player) && topright > 0) total += topright;
        if (this.isCellInBoard(xitl, yitl) && this.isCellValue(xitl, yitl, player) && topleft > 0) total += topleft;
        if (this.isCellInBoard(xibl, yibl) && this.isCellValue(xibl, yibl, player) && bottomleft > 0) total += bottomleft;
        if (this.isCellInBoard(xibr, yibr) && this.isCellValue(xibr, yibr, player) && bottomright > 0) total += bottomright;
 
        // Return 0 if line is not created.
        return total;
    }
 
    /**
     * Performs a move for a specified player
     * @param {number} x The x location of a cell
     * @param {number} y The y location of a cell
     * @param {string} player The player making the move.
     * @throws {InvalidCellLocation} Thrown if the specifed (x, y) cell doesn't exist.
     * @throws {InvalidPlayer} Thrown if the specified player if neither of {'B', 'W'}.
     * @throws {InvalidMove} Thrown if the move is invalid.
     */
    move(x, y, player) {
        if (!this.isCellInBoard(x, y))
            throw "InvalidCellLocation";
        if (!this.validPlayer(player))
            throw "InvalidPlayer";
        if (!this.valid(x, y, player))
            throw "InvalidMove";
        
        // The style of cell to set.
        let opponent = player === 'B' ? 'W' : 'B';
 
        this.setCell(x, y, player);
 
        // The number of opponent cells in each direction.
        let left = 0;
        let right = 0;
        let top = 0;
        let bottom = 0;
        let topleft = 0;
        let topright = 0;
        let bottomleft = 0;
        let bottomright = 0;
 
        // The x and y coordinates to keep track of the cell that is being inspected for each direction.
        let xil = x - 1;
        let xir = x + 1;
        let yit = y - 1;
        let yib = y + 1;
        let xitl = x - 1;
        let yitl = y - 1;
        let xitr = x + 1;
        let yitr = y - 1;
        let xibl = x - 1;
        let yibl = y + 1;
        let xibr = x + 1;
        let yibr = y + 1;
 
        // Count the amount of opponent cells in each direction.
        for (; this.isCellInBoard(xil, y) && this.isCellValue(xil, y, opponent); xil--) left++;
        for (; this.isCellInBoard(x, yit) && this.isCellValue(x, yit, opponent); yit--) top++;
        for (; this.isCellInBoard(xir, y) && this.isCellValue(xir, y, opponent); xir++) right++;
        for (; this.isCellInBoard(x, yib) && this.isCellValue(x, yib, opponent); yib++) bottom++;
        for (; this.isCellInBoard(xitr, yitr) && this.isCellValue(xitr, yitr, opponent); xitr++ , yitr--) topright++;
        for (; this.isCellInBoard(xitl, yitl) && this.isCellValue(xitl, yitl, opponent); xitl-- , yitl--) topleft++;
        for (; this.isCellInBoard(xibr, yibr) && this.isCellValue(xibr, yibr, opponent); xibr++ , yibr++) bottomright++;
        for (; this.isCellInBoard(xibl, yibl) && this.isCellValue(xibl, yibl, opponent); xibl-- , yibl++) bottomleft++;
 
        // Set the chips if the direction is valid to form a line.
        if (this.isCellInBoard(xil, y) && this.isCellValue(xil, y, player) && left > 0) {
            for (let xi = x - 1; this.isCellInBoard(xi, y) && this.isCellValue(xi, y, opponent); xi--) {
                this.setCell(xi, y, player);
            }
        }
 
        if (this.isCellInBoard(xir, y) && this.isCellValue(xir, y, player) && right > 0) {
            for (let xi = x + 1; this.isCellInBoard(xi, y) && this.isCellValue(xi, y, opponent); xi++) {
                this.setCell(xi, y, player);
            }
        }
 
        if (this.isCellInBoard(x, yit) && this.isCellValue(x, yit, player) && top > 0) {
            for (let yi = y - 1; this.isCellInBoard(x, yi) && this.isCellValue(x, yi, opponent); yi--) {
                this.setCell(x, yi, player);
            }
        }
 
        if (this.isCellInBoard(x, yib) && this.isCellValue(x, yib, player) && bottom > 0) {
            for (let yi = y + 1; this.isCellInBoard(x, yi) && this.isCellValue(x, yi, opponent); yi++) {
                this.setCell(x, yi, player);
            }
        }
 
        if (this.isCellInBoard(xitr, yitr) && this.isCellValue(xitr, yitr, player) && topright > 0) {
            for (let xi = x + 1, yi = y - 1; this.isCellInBoard(xi, yi) && this.isCellValue(xi, yi, opponent); xi++ , yi--) {
                this.setCell(xi, yi, player);
            }
        }
 
        if (this.isCellInBoard(xitl, yitl) && this.isCellValue(xitl, yitl, player) && topleft > 0) {
            for (let xi = x - 1, yi = y - 1; this.isCellInBoard(xi, yi) && this.isCellValue(xi, yi, opponent); xi-- , yi--) {
                this.setCell(xi, yi, player);
            }
        }
 
        if (this.isCellInBoard(xibl, yibl) && this.isCellValue(xibl, yibl, player) && bottomleft > 0) {
            for (let xi = x - 1, yi = y + 1; this.isCellInBoard(xi, yi) && this.isCellValue(xi, yi, opponent); xi-- , yi++) {
                this.setCell(xi, yi, player);
            }
        }
 
        if (this.isCellInBoard(xibr, yibr) && this.isCellValue(xibr, yibr, player) && bottomright > 0) {
            for (let xi = x + 1, yi = y + 1; this.isCellInBoard(xi, yi) && this.isCellValue(xi, yi, opponent); xi++ , yi++) {
                this.setCell(xi, yi, player);
            }
        }
 
        // Notify listeners of model update.
        this.notify();
    }
 
    /**
     * Sets the value of a cell.
     * @param {number} x The x location of the cell
     * @param {number} y The y location of the cell
     * @param {string} v The new value of the cell
     * @param {boolean} [notify] Set to true if you want to notify the listeners, otherwise false (also default)
     * @throws {InvalidCellLocation} Thrown if the (x, y) cell doesn't exist.
     * @throws {InvalidCellValue} Thrown if the specified value is neither of {'B', 'W', 'N'}
     */
    setCell(x, y, v, notify) {
        if (!this.isCellInBoard(x, y))
            throw "InvalidCellLocation";
        if (!this.validValue(v))
            throw "InvalidCellValue";
        
        this.state[this.index(x, y)] = v;
 
        if (notify === true)
            this.notify()
    }
 
    /**
     * Gets the value of a cell
     * @param {number} x The x location of the cell
     * @param {number} y The y location of the cell
     * @throws {InvalidCellLocation} Thrown if the (x, y) cell doesn't exist.
     * @return {string} The value which is either {'B', 'W', 'N'} of the specified cell.
     */
    getCell(x, y) {
        if (!this.isCellInBoard(x, y))
            throw "InvalidCellLocation";
        
        return this.state[this.index(x, y)];
    }
 
    /**
     * Gets the count of the player's chip in the board.
     * @param {string} player The player to get the count for.
     * @throws {InvalidPlayer} Thrown if the specified player is neither of {'B', 'W'}
     * @returns {number} The amount of chips the player has on the game board.
     */
    getCount(player) {
        if (!this.validPlayer(player))
            throw "InvalidPlayer";
        
        let sum = 0;
 
        for (let i = 0; i < this.state.length; i++)
            if (this.state[i] === player)
                sum++
        
        return sum;
    }
 
    /**
     * Sets the state of the board.
     * @param {string[]} state The state to set.
     * @throws {InvalidStateSize} Thrown if the specified state is not the correct size.
     * @throws {InvalidCellValue} Thrown if the specified value is neither of {'B', 'W', 'N'}
     */
    setState(state) {
        if (this.size * this.size !== state.length)
            throw "InvalidStateSize";
        
        for (let i = 0; i < state.length; i++)
            if (!this.validValue(state[i]))
                throw "InvalidCellValue";
        
        this.state = JSON.parse(JSON.stringify(state));
 
        // Notify of a state update.
        this.notify();
    }
 
    /**
     * Gets the state of the board.
     * @returns {string[]} An array of {'B', 'W', 'N'} representing the individual cell values.
     */
    getState() {
        return JSON.parse(JSON.stringify(this.state));
    }
 
    /**
     * Sets the state of the board.
     * @param {string} player One of {'B', 'W'} as the player.
     * @throws {InvalidCellValue} Thrown if the specified value is neither of {'B', 'W', 'N'}
     * @returns {number[]} An array of the indices of the cells that are valid.
     */
    findValidMoves(player) {
        if (!this.validPlayer(player))
            throw "InvalidPlayer";
        
        let indicies = [];
 
        for (let y = 0; y < this.size; y++)
            for (let x = 0; x < this.size; x++)
                if (this.valid(x, y, player))
                    indicies.push(this.index(x, y));
 
        return indicies;
    }
 
    /**
     * Adds a listener (a fucntion) to call when the state of the board is updated.
     * @param {function} listener A callback function with no parameters.
     */
    listen(listener) {
        this.listeners.push(listener);
    }
 
    /**
     * Calls the functions that are listening for the change in state.
     */
    notify() {
        for (let listener of this.listeners)
            listener();
    }
 
    /**
     * Removes a listener from the list to notify when there is a state change.
     * @param {function} listener The listener to remove
     */
    ignore(listener) {
        this.listeners = this.listeners.filter(e => e !== listener);
    }
 
    // * Helper and private functions for readability of the code above (not part of the interface).
    index(x, y) {
        return this.size * y + x;
    }
 
    validValue(v) {
        return this.validPlayer(v) || v === 'N';
    }
 
    validPlayer(v) {
        return v === 'B' || v === 'W';
    }
}
