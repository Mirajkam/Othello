var bestMove = 0;
 
/**
 * This is the AI class that provides the functionality of being the opponent to the player.
 */
class AI {
 
    /**
     * Creates an instance of AI.
     * @param {number} difficulty The difficulty of the AI (0 for easy, 1 for average, anything else for hard)
     */
    constructor(difficulty) {
        /**
         * This state variable specified the difficulty of the AI as a number.
         * @property {number} difficulty [difficulty]{@link AI#difficulty}
         */
        this.difficulty = difficulty;
    }
 
    /**
     * Picks the move as the AI based on the set difficulty level.
     * @param {Board} board The board object that represents the current state.
     * @returns {number} The index of the cell to place the AI chip.
     */
    pick(board) {
        if (this.difficulty === 0)
            return this.pickMoveRandom(board.findValidMoves('B'));
        if (this.difficulty === 1)
            return this.pickMoveAverage(board);
 
        return this.pickMoveSmart(board);
    }
 
    /**
     * Picks the move as the ai randomly.
     * @param {number[]} moves The possible moves for the AI.
     * @returns {number} The chosen move.
     * @throws {NoMoves} Thrown if the moves array is of length 0.
     */
    pickMoveRandom(moves) {
        if (moves.length === 0)
            throw "NoMoves";
        
        return moves[Math.floor((Math.random() * moves.length))];
    }
 
    /**
     * Picks the move as the opponent with MiniMax AI algorithm along with the regional heuristic.
     * @param {Board} board The board object that represents the current state.
     * @returns {number} The chosen move index.
     */
    pickMoveAverage(board) {
        let moves = board.findValidMoves('B');
        let size = board.getBoardSize();
        let scores = [];
 
        for (let i of moves)
            scores.push(board.valid(this.x(i, board.getBoardSize()), this.y(i, board.getBoardSize()), 'B'));
 
        let classes = [[], [], [], [], []];
        let scoreclasses = [[], [], [], [], []];
 
        if (moves.length === 0)
            return null;
 
        for (let i = 0; i < moves.length; i++) {
            let x = this.x(moves[i], size);
            let y = this.y(moves[i], size);
 
            let r = this.region(x, y, size);
 
            classes[r - 1].push(moves[i]);
            scoreclasses[r - 1].push(scores[i]);
        }
 
        if (classes[4].length !== 0)
            return classes[4][scoreclasses[4].indexOf(Math.max(...(scoreclasses[4])))];
        else if (classes[2].length !== 0)
            return classes[2][scoreclasses[2].indexOf(Math.max(...(scoreclasses[2])))];
        else if (classes[0].length !== 0)
            return classes[0][scoreclasses[0].indexOf(Math.max(...(scoreclasses[0])))];
        else if (classes[1].length !== 0)
            return classes[1][scoreclasses[1].indexOf(Math.max(...(scoreclasses[1])))];
        else
            return classes[3][scoreclasses[3].indexOf(Math.max(...(scoreclasses[3])))];
    }
 
    /**
     * Picks the move as the opponent with MiniMax AI algorithm along with the regional heuristic.
     * @param {Board} board The board object that represents the current state.
     * @returns {number} The chosen move index.
     */
    pickMoveSmart(board) {
        this.minmax(board, 0, 4, 'B');
        let y = Math.floor(bestMove / board.getBoardSize());
        let x = bestMove % board.getBoardSize();
        return bestMove;
    }
 
    /**
     * Gets the difficulty level set for the ai opponent.
     * @returns The difficulty level (0 for easy, 1 for medium, anything else for hard)
     */
    getDifficulty() {
        return this.difficulty;
    }
 
    /**
     * Sets the difficulty level set for the ai opponent.
     * @param {number} difficulty The difficulty level (0 for easy, 1 for medium, anything else for hard)
     */
    setDifficulty(difficulty) {
        this.difficulty = difficulty;
    }
 
    // * The rest are helper functions for readability or other purposes. Not part of the MIS.
 
    minmax(board, depth, maxDepth, player) {
        var newBoard, score, currMove;
        var moves = board.findValidMoves(player);
 
        // if depth is 0 or if game is over
        if (depth >= maxDepth || moves.length === 0) {
            return board.getCount('B');
        }
 
        let w = [1.5, 1.25, 1.75, 1, 2]
 
        let mscore = (player === 'B' ? - Infinity : Infinity);
 
        for (var i = moves.length - 1; i >= 0; i--) {
            currMove = moves[i];
            newBoard = new Board(board.getBoardSize(), board.getState());
            let y = this.y(currMove, newBoard.getBoardSize())
            let x = this.x(currMove, newBoard.getBoardSize());
            newBoard.move(x, y, player);
 
            score = this.minmax(newBoard, depth + 1, maxDepth, player === 'B' ? 'W' : 'B');
 
            let r = this.region(x, y, newBoard.getBoardSize()) - 1;
 
            score = score * w[r];
 
            if (player === 'B' && mscore < score) {
                mscore = score;
 
                if (depth == 0)
                    bestMove = currMove;
            } else if (player === 'W' && mscore > score) {
                mscore = score;
            }
        }
 
        return mscore;
    }
 
    region(x, y, size) {
        if ((x === 0 || x === size - 1) && (y === 0 || y === size - 1)) {
            return 5;
        } else if (x >= 2 && y >= 2 && x < size - 2 && y < size - 2) {
            return 1;
        } else if (((x === 0 || x === size - 1) && y >= 2 && y < size - 2) || ((y === 0 || y === size - 1) && x >= 2 && x < size - 2)) {
            return 3;
        } else if (((x === 1 || x === size - 2) && y >= 2 && y < size - 2) || ((y === 1 || y === size - 2) && x >= 2 && x < size - 2)) {
            return 2;
        } else {
            return 4;
        }
    }
 
    x(index, size) {
        return index % size;
    }
 
    y(index, size) {
        return Math.floor(index / size);
    }
}
