/**
 * This is the History class that provides the functionality of being able to store history through arrays.
 */
class History {
    /**
     * Creates an instance of History.
     * @param {Object[]} [history] The pre-recorded history data (already existing previous states of the object).
     */
    constructor(history) {
        
        if (history === undefined)
            /**
            * @property {Object[]} data The history data. (The previous states of the object as an array)
            */
            this.data = [];
        else
            this.data = history;
    }
 
    /**
     * Checks if an undo can be performed.
     * @returns {boolean} <code>true</code> if an undo can be performed, otherwise <code>false</code>
     */
    canUndo() {
        return this.data.length > 0;
    }
 
    /**
     * Returns the last state of the data
     * @throws {CannotUndo} Thrown if an undo cannot be performed.
     * @returns {Object} The last state of the data.
     */
    undo() {
        if (!this.canUndo())
            throw "CannotUndo";
        
        return this.data.pop();
    }
 
    /**
     * Resets the state of the history data.
     */
    reset() {
        this.data = [];
    }
 
    /**
     * Adds onto the history of the data.
     * @param {Object} data The data to push onto history.
     */
    push(data) {
        this.data.push(JSON.parse(JSON.stringify(data)));
    }
 
    /**
     * The amount of recorded history.
     * @returns {number} The amount of recorded history.
     */
    time() {
        return this.data.length;
    }
 
    /**
     * View the state at a certain point.
     * @returns {Object} The state at a certain point.
     */
    peek(i) {
        return this.data[i];
    }
 
    /**
     * Get all the recorded history of the object.
     * @returns {Object[]} The previous states of the object as an array.
     */
    timeline() {
        return this.data;
    }
}
